Write a Python function using boto3 that lists all S3 buckets in the current account and returns a dictionary mapping each bucket name to its creation date. Handle any potential ClientError exceptions.
import boto3
from botocore.exceptions import ClientError

def list_s3_buckets() -> dict:
    # Initialize the S3 client
    s3_client = boto3.client('s3')
    result = {}
    try:
        response = s3_client.list_buckets()
        for bucket in response['Buckets']:
            result[bucket['Name']] = bucket['CreationDate']
    except ClientError as e:
        print(f"Error: {e.response['Error']['Message']}")
    return result
